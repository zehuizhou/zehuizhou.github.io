简单爬虫例子

发表于 2018-03-24 | 分类于 爬虫 | 阅读次数

代码如下：

from bs4 import BeautifulSoup
import requests
import pymysql
import os

def get_html(web_url = "https://movie.douban.com/top250?start=%s&filter=1" ): # 爬虫获取网页没啥好说的
header = {
"User-Agent":"Mozilla/5.0 (Windows; U; Windows NT 6.1; en-US) AppleWebKit/534.16 (KHTML, like Gecko) Chrome/10.0.648.133 Safari/534.16"}
html = requests.get(url=web_url, headers=header).text#不加text返回的是response，加了返回的是字符串
Soup = BeautifulSoup(html, "lxml")
data = Soup.find("ol").find_all("li") # 还是有一点要说，就是返回的信息最好只有你需要的那部分，所以这里进行了筛选
return (data)

def get_info(all_move):
for info in all_move:
nums = info.find('em') #排名
num = nums.get_text()

names = info.find("span") #名字名字比较简单 直接获取第一个span就是
name = names.get_text()

charactors = info.find("p") #导演这段信息中有太多非法符号你需要替换掉
charactor = charactors.get_text().replace(" ", "").replace("\n", "") # 使信息排列规律
charactor = charactor.replace("\xa0", "").replace("\xee", "").replace("\xf6", "").replace("\u0161", "").replace(
"\xf4", "").replace("\xfb", "").replace("\u2027", "").replace("\xe5", "")

remarks = info.find_all("span", {"class": "inq"}) #评语
if remarks: # 这个判断是因为有的电影没有评语，你需要做判断
remark = remarks[0].get_text().replace("\u22ef", "")
else:
remark = "此影片没有评价"

scores = info.find_all("span", {"class": "rating_num"})# 评分
score = scores[0].get_text()

#导入数据库
db = pymysql.connect(host="localhost", user="root", password="123456", db="python_test", port=3306,charset="utf8")
cur = db.cursor()

sql_insert ="insert into movie(id,name,charactors,remarks,scores) values('%s','%s','%s','%s','%s')"% \
(num, name, charactor, remark, score)
try:
cur.execute(sql_insert)
# 提交
db.commit()
except Exception as e:
# 错误回滚
db.rollback()
finally:
db.close()

if __name__ == "__main__":
page = 0
while page <= 225:
web_url = "https://movie.douban.com/top250?start=%s&filter=" % page
all_move = get_html(web_url) # 返回每一页的网页
get_info(all_move) # 匹配对应信息存入本地
page += 25
