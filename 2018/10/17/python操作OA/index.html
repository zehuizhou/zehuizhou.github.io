#coding=utf-8
import requests
import datetime
import time
import random

class Punch():
def __init__(self, username, psw):
    self.username = username
    self.psw = psw

def login(self):
    # 获取每次更新的cookie
    url0 = "http://oa.dronggroup.com/ "
    r0 = requests.get(url=url0)
    m = r0.cookies["PHPSESSID"]
    # 登录
    url1 = "http://oa.dronggroup.com/logincheck.php"
    data = {
            "UNAME": self.username,
            "PASSWORD": self.psw,
            "encode_type": 1
            }
    header1 = {
                "Cookie":m,
               }
    r1 = requests.post(url=url1, data=data, headers=header1)
    p = str(r1.cookies).split(",")[0].split()[1]
    p1 = str(r1.cookies).split(",")[1].split()[1]
    p2 = str(r1.cookies).split(",")[2].split()[1]
    p3 = str(r1.cookies).split(",")[3].split()[1]
    u = p + "; " + p1 + "; " + p2 + "; " + p3 + "; "
    # 返回打卡要用到的u
    return u

def on_punch(self):
    nextday = (datetime.datetime.now() + datetime.timedelta(days=1)).strftime("%Y-%m-%d %H:%M:%S")
    on_time = '08:50:00'
    punchtime = nextday[0:11] + on_time  # 上班打卡时间
    timeArray = time.strptime(punchtime, "%Y-%m-%d %H:%M:%S")
    a = int(time.mktime(timeArray))
    b = int(time.time())
    c = a - b + random.randint(0, 300)
    real_punchtime = time.strftime("%Y-%m-%d %H:%M:%S", time.localtime(b + c))
    hour = str(c / 3600)
    print("现在时间：" + datetime.datetime.now().strftime("%F %T"))
    print(str(hour)[0:5] + "小时后打卡")
    print("上班打卡时间：" + real_punchtime)
    time.sleep(c)
    url2 = "http://oa.dronggroup.com/general/attendance/personal/duty/submit.php?REGISTER_TYPE=1"
    header2 = {
               "Cookie": self.login()
                }
    r2 = requests.get(url=url2, headers=header2)

def off_punch(self):
    today = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    off_time = '17:30:00'
    punchtime = today[0:11] + off_time
    timeArray = time.strptime(punchtime, "%Y-%m-%d %H:%M:%S")
    a = int(time.mktime(timeArray))
    b = int(time.time())
    c = a - b + random.randint(0, 60)
    real_punchtime = time.strftime("%Y-%m-%d %H:%M:%S", time.localtime(b + c))
    hour = str(c / 3600)
    print("现在时间：" + datetime.datetime.now().strftime("%F %T"))
    print(str(hour)[0:5] + "小时后打卡")
    print("下班打卡时间：" + real_punchtime)
    time.sleep(c)
    url2 = "http://oa.dronggroup.com/general/attendance/personal/duty/submit.php?REGISTER_TYPE=2"
    header2 = {
               "Cookie": self.login()
                }
    r2 = requests.get(url=url2, headers=header2)

if __name__ == "__main__":
    # 需要抓包提取加密后的密码，之后根据需求调用方法
    p = Punch("zhouzehui", "emhvdTg4NjY4OC4=")
    p.on_punch()
    p.off_punch()
